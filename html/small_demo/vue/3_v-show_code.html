<div id="app">
  <span>第{{i}}页/共{{count}}页</span>
<!--如果i<count就显示按钮-->
  <button v-if="i<count" @click="add">下一页</button>
<!--结果: 如果v-if后的条件满足，则元素显示，否则隐藏元素-->
</div>
<script>
  new Vue({
    el:"#app",
    data:{
      i:1,//当前页码
      count:3, //总页数
    },
    methods:{
      add(){
        this.i++;
      }
    }
  })
</script>
