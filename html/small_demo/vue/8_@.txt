<div id="app">
  <div @click="change">
    <button>-</button>
    <span>{{n}}</span>
    <button>+</button>
  </div>
</div>
<script>
  new Vue({
    el:"#app",
    data:{ n:1 },
    methods:{
      change(e){
        var tar=e.target;
        if(tar.nodeName==="BUTTON"){
          this.n+=tar.innerHTML=="+"?1:-1;
        }
      }
    }
  })
</script>
