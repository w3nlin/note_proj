<div id="app">
  <div style="position:fixed; width:100px; height:100px" :style="divStyle"></div>
</div>
<script>
new Vue({
  el:"#app",
  data:{
  //divStyle:"top:50px; left:50px"
    divStyle:{
      top:"0px",
      left:"0px",
      backgroundColor:"pink"
    }
  },
  watch:{
    //监控内嵌在对象中的变量或属性
    "divStyle.top"(){
      console.log(`top:${this.divStyle.top}`);
    },
    "divStyle.left"(){
      console.log(`left:${this.divStyle.left}`);
    }
  },
  created(){
    setInterval(()=>{
      var left=parseInt(Math.random()*100)+"px",
          top=parseInt(Math.random()*100)+"px",
          backgroundColor=`rgb(${
            parseInt(Math.random()*255)
          },${
            parseInt(Math.random()*255)
          },${
            parseInt(Math.random()*255)
          })`;
      this.divStyle={
        left,top,backgroundColor
      }
    },2000)
  }
})
</script>
