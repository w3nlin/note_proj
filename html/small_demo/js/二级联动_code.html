  <script>
	/*实现“省”和“市”的级联下拉列表*/
	var cities=[
      [{"name":'东城区',"value":101},
       {"name":'西城区',"value":102},
       {"name":'海淀区',"value":103},
       {"name":'朝阳区',"value":104}],
      [{"name":'河东区',"value":201},
       {"name":'河西区',"value":202},
       {"name":'南开区',"value":303}],
      [{"name":'石家庄市',"value":301},
       {"name":'廊坊市',"value":302},
       {"name":'保定市',"value":303},
       {"name":'唐山市',"value":304},
       {"name":'秦皇岛市',"value":304}]
	];
	var provSel=document.getElementsByName("provs")[0];
	var citSel=document.getElementsByName("cities")[0];
	console.log(provSel,citSel);
	provSel.onchange=function(){
	  var i=this.selectedIndex;
		if(i>0){
		  var cts=cities[i-1];
			var frag=document.createDocumentFragment();
			frag.appendChild(new Option("--请选择--"));
			for(var ct of cts){
			   frag.appendChild(new Option(ct.name));
			}
		  citSel.innerHTML="";
			citSel.appendChild(frag);
			citSel.className=""
		}else{
		  citSel.className="hide";
		}
	}
</script>
