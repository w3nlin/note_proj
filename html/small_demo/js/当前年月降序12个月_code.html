      var current_year_temp = new Date().getFullYear();
      var current_month = new Date().getMonth() + 1;
      var date_str ="";
      var date_arr=[];
      
      for(let i=0;i<12;i++){
        var d_value=current_month-i;
        
        if (d_value > 0) {
          d_value = "0" + d_value;
        }
        if(d_value<=0){
          current_year = current_year_temp-1;
          d_value = d_value+12;
          if(d_value!=12&&d_value!=11&&d_value!=10){
            d_value = "0"+d_value;
          }else{
            d_value = "" + d_value;
          }
        }
        date_str = current_year + "-" + d_value + "";
        console.log(date_str);
        date_arr.push(date_str);

      }
