<canvas id="id1" style="background:grey;"></canvas>
<script>
         window.onload=function(){
            var id1=document.getElementById("id1");
            var id11=id1.getContext("2d");

            id1.onmousedown=function(ev){ //鼠标按下的时候
                var ev = ev   //先获取event对象                                   /*|| window.event;*/
                id11.moveTo(ev.clientX-id1.offsetLeft,ev.clientY-id1.offsetTop);//调用moveTo //鼠标的x轴坐标，鼠标的y轴坐标 //减去canvas默认的margin
                

		document.onmousemove=function(ev){ //鼠标移动
                    var ev=ev   //同样获取event对象                     /*||window.event;*/
                    id11.lineTo(ev.clientX-id1.offsetLeft,ev.clientY-id1.offsetTop); //移动时鼠标的点对应的就是lineTo（）的点
                    id11.stroke();//画线
                };
                document.onmouseup=function(){ //鼠标抬起的时候将两个事件取消掉
                    document.onmousemove=null;
                    document.onmouseup=null;
                  };
            };

        }

    </script>
